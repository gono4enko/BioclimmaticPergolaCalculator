  def get_pergola_count():

 БЛОК 9 — ПАРСЕР ИЗОБРАЖЕНИЙ С DECOLIFE
════════════════════════════════════════

Создать/обновить scripts/fetch_decolife.py:

  import requests, json, os
  from bs4 import BeautifulSoup
  from urllib.parse import urljoin

  HEADERS = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"}

  SOURCES = {
    'b500': 'https://decolife.pro/products/pergolas/bioclimatic-pergolas/bioclimatic-with-rotating-lamellas/',
    'b700': 'https://decolife.pro/products/pergolas/bioclimatic-pergolas/bioclimatic-with-sliding-lamellas/',
    'b600': 'https://decolife.pro/products/pergolas/bioclimatic-pergolas/bioclimatic-with-pir-sandwich-panels/',
  }

  def fetch_model(model_key, url):
      os.makedirs(f'static/decolife/{model_key}', exist_ok=True)
      r = requests.get(url, headers=HEADERS, timeout=15)
      soup = BeautifulSoup(r.text, 'lxml')

      # Описание
      title = soup.find('h1')
      title_text = title.get_text(strip=True) if title else ''
      main = soup.find('main') or soup.find('article') or soup.body
      paras = [p.get_text(strip=True) for p in main.find_all('p') if len(p.get_text()) > 60]
      description = paras[0] if paras else ''
      features = [li.get_text(strip=True) for li in main.find_all('li') if len(li.get_text()) > 15][:8]

      # Изображения
      imgs_saved = []
      count = 0
      for img in main.find_all('img'):
          if count >= 6:
              break
          src = img.get('src') or img.get('data-src', '')
          if not src:
              continue
          # Пропускать иконки, логотипы, флаги
          skip_words = ['icon','logo','flag','arrow','close','menu','social']
          if any(w in src.lower() for w in skip_words):
              continue
          if not any(src.lower().endswith(ext) for ext in ['.jpg','.jpeg','.png','.webp']):
              continue
          abs_url = urljoin('https://decolife.pro', src)
          try:
              img_r = requests.get(abs_url, headers=HEADERS, timeout=10)
              if len(img_r.content) < 10000:  # пропускать маленькие (иконки)
                  continue
              fname = f'img_{count}.jpg'
              fpath = f'static/decolife/{model_key}/{fname}'
              with open(fpath, 'wb') as f:
                  f.write(img_r.content)
              imgs_saved.append({'url': abs_url, 'local': fname, 'alt': img.get('alt','')})
              count += 1
          except Exception:
              continue

      data = {
          'model': model_key.upper(),
          'title': title_text,
          'description': description,
          'features': features,
          'images': imgs_saved,
          'cover': imgs_saved[0]['local'] if imgs_saved else None,
      }
      with open(f'static/decolife/{model_key}/data.json', 'w', encoding='utf-8') as f:
          json.dump(data, f, ensure_ascii=False, indent=2)
      print(f'[OK] {model_key}: {len(imgs_saved)} фото, описание {len(description)} симв.')

  def fetch_all():
      for key, url in SOURCES.items():
          try:
              fetch_model(key, url)
          except Exception as e:
              print(f'[ERR] {key}: {e}')

  if __name__ == '__main__':
      fetch_all()

Запуск при старте app.py:
  import os
  if not os.path.exists('static/decolife/b600'):
      from scripts.fetch_decolife import fetch_all
      fetch_all()

Использование в шаблонах:
  Загружать data.json для модели → показывать:
  - cover: первое фото в шапке блока «О перголе от производителя»
  - description: текст описания
  - features: список преимуществ (3–5 пунктов)
  - Подпись: «Описание и фото: decolife.pro — официальный производитель»

  Fallback если static/decolife не заполнен:
  → использовать COVER_IMAGES и GALLERY_BY_MODEL из Блока 1 и 2

  import qrcode, io, base64

  def generate_qr_base64(url='https://pergolamarket.ru'):
      qr = qrcode.QRCode(box_size=4, border=2)
      qr.add_data(url)
      qr.make(fit=True)
      img = qr.make_image(fill_color='black', back_color='white')
      buf = io.BytesIO()
      img.save(buf, format='PNG')
      return base64.b64encode(buf.getvalue()).decode()

 БЛОК 12 — ПОСТОЯННАЯ ССЫЛКА НА КП
════════════════════════════════════════

При нажатии «Рассчитать» — сохранять параметры расчёта.

Вариант реализации (простой, без БД):
  import json, os, random, string
  from datetime import datetime

  def save_calculation(data: dict) -> str:
      os.makedirs('data/calculations', exist_ok=True)
      calc_id = ''.join(random.choices(string.ascii_lowercase + string.digits, k=8))
      data['saved_at'] = datetime.now().isoformat()
      with open(f'data/calculations/{calc_id}.json', 'w') as f:
          json.dump(data, f, ensure_ascii=False)
      return calc_id

  Маршрут для загрузки КП по ссылке:
  @app.route('/kp/')
  def view_kp(calc_id):
      path = f'data/calculations/{calc_id}.json'
      if not os.path.exists(path):
          return 'КП не найдено', 404
      with open(path) as f:
          data = json.load(f)
      return render_template('kp.html', **data,
                             pergola_count=get_pergola_count(),
                             current_year=get_current_year())
