    participant U as Пользователь

    U->>F: Нажатие "Экспорт PDF"

    A->>P: Генерация PDF

    A->>F: HTTP Response (PDF)
    F->>B: Отображение/скачивание

# app.py
import os
import logging
from datetime import datetime
from flask import Flask, request, make_response, jsonify
from urllib.parse import quote
import pytz
@app.route('/export_pdf_new', methods=['POST'])
def export_pdf_new():
    """
    Основной маршрут для экспорта PDF коммерческих предложений
    
    Принимает: JSON с данными расчета
    Возвращает: PDF файл с корректными заголовками
    """
    try:
        # Логирование начала обработки
        logging.info("📄 Получен запрос на экспорт PDF")
        
        # 1. Получение и валидация данных
        request_data = request.get_json()
        if not request_data:
            return jsonify({
                "success": False,
                "error": "Отсутствуют данные для генерации PDF"
            }), 400
        
        # 2. Валидация обязательных полей
        required_fields = ['constructionType', 'model', 'width', 'depth']
        missing_fields = [field for field in required_fields if field not in request_data]
        
        if missing_fields:
            return jsonify({
                "success": False,
                "error": f"Отсутствуют обязательные поля: {', '.join(missing_fields)}"
            }), 400
        
        # 3. Подготовка данных для PDF генератора
        pdf_data = prepare_pdf_data(request_data)
        
        # 4. Генерация PDF документа
        from simplified_pdf_generator import generate_pdf
        pdf_bytes = generate_pdf(pdf_data)
        
        if not pdf_bytes:
            return jsonify({
                "success": False,
                "error": "Ошибка при генерации PDF документа"
            }), 500
        
        # 5. Формирование имени файла
        filename = create_pdf_filename(pdf_data)
        
        # 6. Создание HTTP ответа с PDF
        response = make_response(pdf_bytes)
        
        # 7. Установка заголовков для корректной обработки браузером
        setup_pdf_headers(response, filename, len(pdf_bytes))
        
        logging.info(f"✅ PDF успешно сгенерирован: {filename} ({len(pdf_bytes)/1024:.1f} КБ)")
        
        return response
        
    except ValidationError as ve:
        logging.warning(f"⚠️ Ошибка валидации: {ve}")
        return jsonify({
            "success": False,
            "error": f"Ошибка валидации данных: {str(ve)}"
        }), 400
        
    except Exception as e:
        logging.error(f"❌ Критическая ошибка при генерации PDF: {e}")
        
        # Возвращаем PDF с ошибкой вместо JSON
        error_pdf = generate_error_pdf(str(e))
        response = make_response(error_pdf)
        setup_pdf_headers(response, "error_document.pdf", len(error_pdf))
        
        return response
def prepare_pdf_data(request_data):
    """
    Подготавливает данные для PDF генератора
    
    Args:
        request_data (dict): Данные из запроса
        
    Returns:
        dict: Подготовленные данные для PDF
    """
    try:
        # Базовая информация о конструкции
        construction_info = {
            'type': get_construction_type_display(request_data['constructionType']),
            'model': request_data.get('model', 'Basic & Pro New'),
            'width': float(request_data.get('width', 0)),
            'depth': float(request_data.get('depth', 0)),
            'color': request_data.get('color', 'Белый (RAL 9010)'),
            'area': float(request_data.get('width', 0)) * float(request_data.get('depth', 0))
        }
        
        # Ценовая информация
        pricing_info = {
            'base_price': float(request_data.get('basePrice', 0)),
            'options_price': float(request_data.get('optionsPrice', 0)),
            'color_surcharge': float(request_data.get('colorSurcharge', 0)),
            'total_price': float(request_data.get('totalPrice', 0))
        }
        
        # Опции и дополнения
        options_info = request_data.get('options', {})
        options_details = request_data.get('optionsDetails', {})
        
        return {
            'construction': construction_info,
            'pricing': pricing_info,
            'options': options_info,
            'options_details': options_details,
            'metadata': {
                'creation_date': datetime.now(pytz.timezone('Europe/Moscow')),
                'source': 'calculator'
            }
        }
        
    except Exception as e:
        logging.error(f"Ошибка при подготовке данных PDF: {e}")
        raise ValidationError(f"Некорректные данные расчета: {e}")
def create_pdf_filename(pdf_data):
    """
    Создает информативное имя файла PDF
    
    Args:
        pdf_data (dict): Данные PDF
        
    Returns:
        str: Имя файла
    """
    try:
        construction = pdf_data['construction']
        creation_date = pdf_data['metadata']['creation_date']
        
        # Формируем базовое имя
        base_name = (
            f"КП тентовая пергола "
            f"{construction['type']} "
            f"{construction['model']} "
            f"{construction['width']} м × {construction['depth']} м "
            f"{creation_date.strftime('%d.%m.%Y')}"
        )
        
        # Очищаем от недопустимых символов
        clean_name = sanitize_filename(base_name)
        
        return f"{clean_name}.pdf"
        
    except Exception as e:
        logging.error(f"Ошибка при создании имени файла: {e}")
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        return f"commercial_proposal_{timestamp}.pdf"
def setup_pdf_headers(response, filename, file_size):
    """
    Настраивает HTTP заголовки для PDF ответа
    
    Args:
        response: Flask response object
        filename (str): Имя файла
        file_size (int): Размер файла в байтах
    """
    # Основные заголовки
    response.headers['Content-Type'] = 'application/pdf'
    response.headers['Content-Length'] = str(file_size)
    
    # Content-Disposition с поддержкой Unicode
    ascii_filename = "commercial_proposal.pdf"  # Fallback для старых браузеров
    unicode_filename = quote(filename)  # URL-encoded Unicode имя
    
    response.headers['Content-Disposition'] = (
        f'inline; '
        f'filename="{ascii_filename}"; '
        f'filename*=UTF-8\'\'{unicode_filename}'
    )
    
    # Дополнительные заголовки для кэширования
    response.headers['Cache-Control'] = 'no-cache, no-store, must-revalidate'
    response.headers['Pragma'] = 'no-cache'
    response.headers['Expires'] = '0'
    
    logging.info(f"Установлены заголовки PDF: {filename}")
def sanitize_filename(filename):
    """
    Очищает имя файла от недопустимых символов
    
    Args:
        filename (str): Исходное имя файла
        
    Returns:
        str: Очищенное имя файла
    """
    import re
    
    # Заменяем недопустимые символы
    invalid_chars = r'[\/:*?"<>|]'
    clean_name = re.sub(invalid_chars, '_', filename)
    
    # Заменяем множественные пробелы и подчеркивания
    clean_name = re.sub(r'\s+', ' ', clean_name)
    clean_name = re.sub(r'_+', '_', clean_name)
    
    # Убираем начальные и конечные пробелы
    clean_name = clean_name.strip(' _')
    
    # Ограничиваем длину
    if len(clean_name) > 200:
        clean_name = clean_name[:200]
    
    return clean_name
class ValidationError(Exception):

def get_construction_type_display(construction_type):
    """
    Преобразует код типа конструкции в читаемое название
    
    Args:
        construction_type (str): Код типа конструкции
        
    Returns:
        str: Читаемое название
    """
    type_mapping = {
        'wall': 'Пристенная',
        'standalone': 'Отдельностоящая',
        'twin': 'Двухскатная',
        'console': 'Консольная',
        'wall_fly': 'Пристенная FLY',
        'solo_fly': 'Отдельностоящая FLY',
        'twin_fly': 'Двухскатная FLY'
    }
    
    return type_mapping.get(construction_type, construction_type)
def generate_error_pdf(error_message):
    """
    Генерирует PDF документ с сообщением об ошибке
    
    Args:
        error_message (str): Текст ошибки
        
    Returns:
        bytes: PDF документ с ошибкой
    """
    try:
        from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
        from reportlab.lib.styles import getSampleStyleSheet
        from reportlab.lib.pagesizes import A4
        from io import BytesIO
        
        buffer = BytesIO()
        doc = SimpleDocTemplate(buffer, pagesize=A4)
        styles = getSampleStyleSheet()
        
        elements = [
            Paragraph("Ошибка генерации коммерческого предложения", styles['Title']),
            Spacer(1, 20),
            Paragraph(f"Описание ошибки: {error_message}", styles['Normal']),
            Spacer(1, 10),
            Paragraph("Пожалуйста, попробуйте еще раз или обратитесь в техническую поддержку.", styles['Normal']),
            Spacer(1, 10),
            Paragraph(f"Время ошибки: {datetime.now().strftime('%d.%m.%Y %H:%M:%S')}", styles['Normal'])
        ]
        
        doc.build(elements)
        return buffer.getvalue()
        
    except Exception:
        # Минимальный PDF в случае критической ошибки
        return create_minimal_error_pdf()

// static/js/pdf-exporter.js
class PDFExporter {
    constructor() {
        this.isExporting = false;
        this.apiEndpoint = '/export_pdf_new';
        this.init();
    }
    
    init() {
        this.setupExportButtons();
        this.injectStyles();
    }
    
    /**
     * Настраивает кнопки экспорта PDF
     */

        const exportButtons = document.querySelectorAll('.pdf-export-btn, #exportPdfBtn');

            button.addEventListener('click', (e) => {
                e.preventDefault();
                this.handleExportClick();
            });
        });
        
        // Если кнопок нет, создаем их
        if (exportButtons.length === 0) {
            this.createExportButton();
        }
    }
    
    /**
     * Создает кнопку экспорта PDF
     */
    createExportButton() {

        } else {
            document.body.appendChild(button);
        }
    }
    
    /**
     * Обрабатывает клик по кнопке экспорта
     */
    async handleExportClick() {
        if (this.isExporting) {
            return; // Предотвращаем множественные запросы
        }
        
        try {
            this.isExporting = true;
            this.showLoadingState();
            
            // Собираем данные для экспорта
            const exportData = this.gatherCalculationData();
            
            if (!exportData) {
                throw new Error('Нет данных для экспорта. Выполните расчет перголы.');
            }
            
            console.log('📊 Отправляем данные для экспорта:', exportData);
            
            // Отправляем запрос на сервер
            const response = await fetch(this.apiEndpoint, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(exportData)
            });
            
            if (!response.ok) {
                // Пытаемся получить JSON ошибку
                let errorMessage;
                try {
                    const errorData = await response.json();
                    errorMessage = errorData.error || `Ошибка сервера: ${response.status}`;
                } catch {
                    errorMessage = `HTTP ошибка: ${response.status} ${response.statusText}`;
                }
                throw new Error(errorMessage);
            }
            
            // Проверяем тип содержимого
            const contentType = response.headers.get('content-type');
            if (!contentType || !contentType.includes('application/pdf')) {
                throw new Error('Сервер не вернул PDF документ');
            }
            
            // Извлекаем имя файла из заголовков
            const fileName = this.extractFilename(response.headers);
            
            // Получаем PDF как blob
            const blob = await response.blob();
            
            // Создаем URL для blob
            const blobUrl = URL.createObjectURL(blob);
            
            console.log(`📄 PDF успешно получен: ${fileName} (${blob.size} байт)`);
            
            // Показываем опции для работы с PDF
            this.showPDFOptions(blobUrl, fileName);
            
        } catch (error) {
            console.error('❌ Ошибка при экспорте PDF:', error);
            this.showError(error.message);
        } finally {
            this.isExporting = false;
            this.hideLoadingState();
        }
    }
    
    /**
     * Собирает данные расчета для экспорта
     */
    gatherCalculationData() {
        // Приоритет источников данных
        const dataSources = [
            () => this.getDataFromGlobalVariable(),
            () => this.getDataFromLocalStorage(),
            () => this.getDataFromForm(),
            () => this.getDataFromResults()
        ];
        
        for (const source of dataSources) {
            try {
                const data = source();
                if (data && this.validateCalculationData(data)) {
                    console.log('✅ Данные получены из источника:', source.name);
                    return data;
                }
            } catch (error) {
                console.warn(`⚠️ Ошибка получения данных из ${source.name}:`, error);
            }
        }
        
        return null;
    }
    
    /**
     * Получает данные из глобальной переменной
     */
    getDataFromGlobalVariable() {
        // Проверяем различные глобальные переменные
        const globalSources = [
            'window.lastCalculationData',
            'window.calculationResult',
            'window.pergolaData'
        ];
        
        for (const source of globalSources) {
            try {
                const data = eval(source);
                if (data && typeof data === 'object') {
                    return data;
                }
            } catch (error) {
                // Игнорируем ошибки eval
            }
        }
        
        return null;
    }
    
    /**
     * Получает данные из localStorage
     */
    getDataFromLocalStorage() {
        const storageKeys = [
            'lastCalculation',
            'pergolaCalculation',
            'calculationData'
        ];
        
        for (const key of storageKeys) {
            try {
                const stored = localStorage.getItem(key);
                if (stored) {
                    return JSON.parse(stored);
                }
            } catch (error) {
                console.warn(`Ошибка чтения localStorage[${key}]:`, error);
            }
        }
        
        return null;
    }
    
    /**
     * Получает данные из формы калькулятора
     */
    getDataFromForm() {
        const form = document.querySelector('#calculatorForm, .calculator-form, form');
        if (!form) return null;
        
        const formData = new FormData(form);
        const data = {};
        
        // Преобразуем FormData в объект
        for (let [key, value] of formData.entries()) {
            data[key] = value;
        }
        
        // Добавляем данные из скрытых полей и элементов
        const additionalFields = [
            'constructionType', 'model', 'width', 'depth', 'color',
            'basePrice', 'optionsPrice', 'totalPrice', 'numConstructions'
        ];
        
        additionalFields.forEach(field => {
            const element = document.querySelector(`[name="${field}"], #${field}, .${field}`);
            if (element) {
                if (element.type === 'checkbox') {
                    data[field] = element.checked;
                } else if (element.type === 'radio') {
                    if (element.checked) {
                        data[field] = element.value;
                    }
                } else {
                    data[field] = element.value;
                }
            }
        });
        
        return Object.keys(data).length > 0 ? data : null;
    }
    
    /**
     * Получает данные из результатов на странице
     */
    getDataFromResults() {
        const data = {};
        
        // Ищем элементы с результатами
        const resultElements = {
            totalPrice: '.total-price, .final-price, .result-total',
            basePrice: '.base-price, .construction-price',
            constructionType: '.construction-type, .pergola-type',
            model: '.model-name, .pergola-model',
            width: '.width-value, .pergola-width',
            depth: '.depth-value, .pergola-depth'
        };
        
        for (const [key, selectors] of Object.entries(resultElements)) {
            for (const selector of selectors.split(', ')) {
                const element = document.querySelector(selector);
                if (element) {
                    let value = element.textContent.trim();
                    
                    // Очищаем числовые значения
                    if (['totalPrice', 'basePrice', 'width', 'depth'].includes(key)) {
                        const numbers = value.match(/[\d\s,]+/);
                        if (numbers) {
                            value = parseFloat(numbers[0].replace(/[\s,]/g, ''));
                        }
                    }
                    
                    data[key] = value;
                    break;
                }
            }
        }
        
        return Object.keys(data).length > 0 ? data : null;
    }
    
    /**
     * Валидирует данные расчета
     */
    validateCalculationData(data) {
        const requiredFields = ['constructionType', 'width', 'depth'];
        
        for (const field of requiredFields) {
            if (!data[field]) {
                console.warn(`❌ Отсутствует обязательное поле: ${field}`);
                return false;
            }
        }
        
        // Проверяем числовые значения
        if (isNaN(parseFloat(data.width)) || isNaN(parseFloat(data.depth))) {
            console.warn('❌ Некорректные размеры перголы');
            return false;
        }
        
        return true;
    }
    
    /**
     * Извлекает имя файла из заголовков HTTP
     */
    extractFilename(headers) {
        const contentDisposition = headers.get('content-disposition');
        if (!contentDisposition) {
            return 'commercial_proposal.pdf';
        }
        
        // Пробуем извлечь Unicode имя
        const filenameStarMatch = contentDisposition.match(/filename\*=UTF-8''([^;]+)/i);
        if (filenameStarMatch) {
            try {
                return decodeURIComponent(filenameStarMatch[1]);
            } catch (e) {
                console.warn('Ошибка декодирования имени файла:', e);
            }
        }
        
        // Пробуем обычный параметр filename
        const filenameMatch = contentDisposition.match(/filename=["']?([^"';\s]+)/i);
        if (filenameMatch) {
            return filenameMatch[1];
        }
        
        return 'commercial_proposal.pdf';
    }
    
    /**
     * Показывает опции для работы с PDF
     */
    showPDFOptions(blobUrl, fileName) {
        // Удаляем предыдущие модальные окна
        const existingModal = document.querySelector('.pdf-modal');
        if (existingModal) {
            existingModal.remove();
        }

// static/js/iframe-integration.js
class IframeIntegration {
    constructor() {
        this.isInIframe = window.self !== window.top;
        this.parentOrigin = null;
        this.heightObserver = null;
        
        if (this.isInIframe) {
            this.init();
        }
    }
    
    init() {
        this.setupMessageListener();
        this.setupHeightObserver();
        this.setupPDFExportHandler();
        
        console.log('🖼️ Калькулятор запущен в iframe');
    }
    
    setupMessageListener() {
        window.addEventListener('message', (event) => {
            // Сохраняем origin родительского окна
            if (!this.parentOrigin) {
                this.parentOrigin = event.origin;
            }
            
            const data = event.data;
            
            switch (data.type) {
                case 'PARENT_READY':
                    this.handleParentReady(data.config);
                    break;
                    
                case 'CONFIG_UPDATE':
                    this.handleConfigUpdate(data.config);
                    break;
            }
        });
    }
    
    setupHeightObserver() {
        // Отслеживаем изменения высоты контента
        const targetNode = document.body;
        
        const observer = new ResizeObserver((entries) => {
            for (let entry of entries) {
                const height = entry.contentRect.height;
                this.sendHeightUpdate(height);
            }
        });
        
        observer.observe(targetNode);
        this.heightObserver = observer;
        
        // Дополнительно отслеживаем изменения DOM
        const mutationObserver = new MutationObserver(() => {
            setTimeout(() => {
                this.sendHeightUpdate(document.body.scrollHeight);
            }, 100);
        });
        
        mutationObserver.observe(document.body, {
            childList: true,
            subtree: true,
            attributes: true
        });
    }
    
    setupPDFExportHandler() {
        // Перехватываем экспорт PDF для отправки в родительское окно
        if (window.pdfExporter) {
            const originalHandleExportClick = window.pdfExporter.handleExportClick;
            
            window.pdfExporter.handleExportClick = async function() {
                // Отправляем уведомление в родительское окно
                if (this.isInIframe) {
                    this.sendMessage({
                        type: 'PDF_EXPORT_START',
                        timestamp: Date.now()
                    });
                }
                
                try {
                    // Выполняем оригинальную функцию
                    await originalHandleExportClick.call(window.pdfExporter);
                    
                    // Уведомляем о успешном экспорте
                    if (this.isInIframe) {
                        this.sendMessage({
                            type: 'PDF_EXPORT_SUCCESS',
                            timestamp: Date.now()
                        });
                    }
                } catch (error) {
                    // Уведомляем об ошибке
                    if (this.isInIframe) {
                        this.sendMessage({
                            type: 'PDF_EXPORT_ERROR',
                            error: error.message,
                            timestamp: Date.now()
                        });
                    }
                    throw error;
                }
            }.bind(this);
        }
    }
    
    sendHeightUpdate(height) {
        this.sendMessage({
            type: 'HEIGHT_UPDATE',
            height: height,
            timestamp: Date.now()
        });
    }
    
    sendCalculationComplete(calculationData) {
        this.sendMessage({
            type: 'CALCULATION_COMPLETE',
            calculationData: calculationData,
            timestamp: Date.now()
        });
    }
    
    sendMessage(data) {
        if (this.parentOrigin && window.parent) {
            window.parent.postMessage(data, this.parentOrigin);
        }
    }
    
    handleParentReady(config) {
        console.log('👨‍👩‍👧‍👦 Родительское окно готово, конфигурация:', config);
        
        // Применяем конфигурацию от родительского окна
        if (config.theme === 'tilda') {
            document.body.classList.add('tilda-theme');
        }
        
        // Отправляем начальную высоту
        this.sendHeightUpdate(document.body.scrollHeight);
    }
    
    handleConfigUpdate(config) {
        console.log('⚙️ Обновление конфигурации:', config);
        // Обрабатываем обновления конфигурации
    }
}

def setup_pdf_response_headers(response, filename, file_size):
    """
    Настраивает все необходимые HTTP заголовки для PDF ответа
    
    Args:
        response: Flask response object
        filename (str): Имя файла PDF
        file_size (int): Размер файла в байтах
    """
    # Основные заголовки
    response.headers['Content-Type'] = 'application/pdf'
    response.headers['Content-Length'] = str(file_size)
    
    # Content-Disposition с поддержкой Unicode
    from urllib.parse import quote
    
    # ASCII fallback для старых браузеров
    ascii_filename = "document.pdf"
    
    # URL-encoded Unicode имя для современных браузеров
    unicode_filename = quote(filename)
    
    # Комбинированный заголовок
    response.headers['Content-Disposition'] = (
        f'inline; '
        f'filename="{ascii_filename}"; '
        f'filename*=UTF-8\'\'{unicode_filename}'
    )
    
    # Заголовки кэширования
    response.headers['Cache-Control'] = 'no-cache, no-store, must-revalidate'
    response.headers['Pragma'] = 'no-cache'
    response.headers['Expires'] = '0'
    
    # Заголовки безопасности
    response.headers['X-Content-Type-Options'] = 'nosniff'
    response.headers['X-Frame-Options'] = 'SAMEORIGIN'
    
    # CORS заголовки (если нужно)
    if request.headers.get('Origin'):
        response.headers['Access-Control-Allow-Origin'] = request.headers['Origin']
        response.headers['Access-Control-Allow-Credentials'] = 'true'
    
    return response

// Детекция браузера и адаптация поведения
class BrowserCompatibility {
    static detectBrowser() {
        const userAgent = navigator.userAgent;
        
        if (userAgent.includes('Chrome')) {
            return 'chrome';
        } else if (userAgent.includes('Firefox')) {
            return 'firefox';
        } else if (userAgent.includes('Safari') && !userAgent.includes('Chrome')) {
            return 'safari';
        } else if (userAgent.includes('Edge')) {
            return 'edge';
        } else {
            return 'other';
        }
    }
    
    static adaptPDFHandling(blobUrl, fileName) {
        const browser = this.detectBrowser();
        
        switch (browser) {
            case 'safari':
                // Safari требует особой обработки
                return this.handleSafariPDF(blobUrl, fileName);
                
            case 'firefox':
                // Firefox хорошо работает с blob URLs
                return this.handleStandardPDF(blobUrl, fileName);
                
            case 'chrome':
            case 'edge':
                // Chrome и Edge поддерживают все современные функции
                return this.handleStandardPDF(blobUrl, fileName);
                
            default:
                // Консервативный подход для неизвестных браузеров
                return this.handleLegacyPDF(blobUrl, fileName);
        }
    }
    
    static handleSafariPDF(blobUrl, fileName) {
        // Safari может некорректно обрабатывать Unicode имена файлов
        // Используем метод через создание ссылки
        const link = document.createElement('a');
        link.href = blobUrl;
        link.download = fileName.replace(/[^\w\s.-]/g, '_'); // Упрощаем имя файла
        link.style.display = 'none';
        
        document.body.appendChild(link);
        link.click();
        
        setTimeout(() => {
            document.body.removeChild(link);
            URL.revokeObjectURL(blobUrl);
        }, 100);
        
        return true;
    }
    
    static handleStandardPDF(blobUrl, fileName) {
        // Стандартная обработка для современных браузеров
        const link = document.createElement('a');
        link.href = blobUrl;
        link.download = fileName;
        link.style.display = 'none';
        
        document.body.appendChild(link);
        link.click();
        
        setTimeout(() => {
            document.body.removeChild(link);
            URL.revokeObjectURL(blobUrl);
        }, 100);
        
        return true;
    }
    
    static handleLegacyPDF(blobUrl, fileName) {
        // Консервативный подход для старых браузеров
        try {
            // Пробуем стандартный метод
            if (this.handleStandardPDF(blobUrl, fileName)) {
                return true;
            }
        } catch (error) {
            console.warn('Стандартный метод не работает, используем fallback');
        }
        
        // Fallback: открываем в новой вкладке
        window.open(blobUrl, '_blank');
        
        setTimeout(() => {
            URL.revokeObjectURL(blobUrl);
        }, 5000);
        
        return true;
    }
}

class PDFExportDebugger:
    """Класс для отладки процесса экспорта PDF"""
    
    def __init__(self):
        self.logs = []
        self.start_time = None
        
    def start_debug_session(self):
        """Начинает сессию отладки"""
        self.start_time = time.time()
        self.logs = []
        self.log("DEBUG_SESSION_START", "Начата сессия отладки PDF экспорта")
        
    def log(self, event_type, message, data=None):
        """Записывает событие в лог отладки"""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'elapsed': time.time() - self.start_time if self.start_time else 0,
            'event_type': event_type,
            'message': message,
            'data': data
        }
        
        self.logs.append(log_entry)
        logging.debug(f"[PDF_DEBUG] {event_type}: {message}")
        
    def log_request_data(self, request_data):
        """Логирует данные запроса"""
        self.log("REQUEST_DATA", "Получены данные запроса", {
            'data_size': len(str(request_data)),
            'keys': list(request_data.keys()) if isinstance(request_data, dict) else None,
            'sample_data': str(request_data)[:200] + '...' if len(str(request_data)) > 200 else str(request_data)
        })
        
    def log_validation_result(self, is_valid, errors=None):
        """Логирует результат валидации"""
        self.log("VALIDATION_RESULT", 
                f"Валидация {'прошла' if is_valid else 'не прошла'}", 
                {'is_valid': is_valid, 'errors': errors})
                
    def log_pdf_generation(self, pdf_size, generation_time):
        """Логирует результат генерации PDF"""
        self.log("PDF_GENERATION", 
                f"PDF сгенерирован за {generation_time:.2f}с", 
                {'pdf_size': pdf_size, 'generation_time': generation_time})
                
    def log_error(self, error_type, error_message, traceback_info=None):
        """Логирует ошибку"""
        self.log("ERROR", f"{error_type}: {error_message}", {
            'error_type': error_type,
            'error_message': error_message,
            'traceback': traceback_info
        })
        
    def get_debug_report(self):
        """Возвращает отчет отладки"""
        total_time = time.time() - self.start_time if self.start_time else 0
        
        return {
            'session_info': {
                'start_time': self.start_time,
                'total_duration': total_time,
                'events_count': len(self.logs)
            },
            'events': self.logs,
            'summary': self._generate_summary()
        }
        
    def _generate_summary(self):
        """Генерирует сводку по сессии отладки"""
        event_types = {}
        errors = []
        
        for log_entry in self.logs:
            event_type = log_entry['event_type']
            event_types[event_type] = event_types.get(event_type, 0) + 1
            
            if event_type == 'ERROR':
                errors.append(log_entry)
                
        return {
            'event_types': event_types,
            'errors_count': len(errors),
            'errors': errors
        }

class PDFExportMonitor:

            'pdf_size': pdf_size,
            'success': success

            'error_type': error_type,
            'error_message': error_message

    def get_health_status(self):
        """Возвращает статус здоровья системы"""
        now = time.time()
        last_hour = now - 3600
        
        # Метрики за последний час
        recent_requests = [r for r in self.metrics['requests'] if r['timestamp'] > last_hour]
        recent_generations = [g for g in self.metrics['generations'] if g['timestamp'] > last_hour]
        recent_errors = [e for e in self.metrics['errors'] if e['timestamp'] > last_hour]
        
        # Вычисляем показатели
        total_requests = len(recent_requests)
        successful_generations = len([g for g in recent_generations if g['success']])
        success_rate = (successful_generations / total_requests) if total_requests > 0 else 1.0
        
        avg_response_time = (
            sum(g['duration'] for g in recent_generations) / len(recent_generations)
            if recent_generations else 0
        )
        
        # Системные ресурсы
        system_stats = {
            'cpu_percent': psutil.cpu_percent(interval=1),
            'memory_percent': psutil.virtual_memory().percent,
            'disk_percent': psutil.disk_usage('/').percent
        }
        
        health_status = 'healthy'
        if success_rate < 0.95:
            health_status = 'degraded'
        if success_rate < 0.8 or len(recent_errors) > 10:
            health_status = 'unhealthy'
            
        return {
            'status': health_status,
            'uptime': now - self.start_time,
            'metrics': {
                'requests_last_hour': total_requests,
                'success_rate': success_rate,
                'avg_response_time': avg_response_time,
                'errors_last_hour': len(recent_errors)
            },
            'system': system_stats,
            'alerts': self.get_active_alerts()
        }
        
    def _check_required_fields(self, request_data):
        """Проверяет наличие обязательных полей"""
        required_fields = ['constructionType', 'width', 'depth']
        return all(field in request_data for field in required_fields)
        
    def _check_performance_alerts(self, duration, pdf_size):
        """Проверяет на проблемы с производительностью"""
        if duration > 5.0:  # Более 5 секунд
            self._add_alert('SLOW_GENERATION', f'Медленная генерация PDF: {duration:.2f}с')
            
        if pdf_size > 10 * 1024 * 1024:  # Более 10 МБ
            self._add_alert('LARGE_PDF', f'Большой размер PDF: {pdf_size/1024/1024:.1f} МБ')
            
    def _check_error_alerts(self, error_type):
        """Проверяет на критические ошибки"""
        error_counts = defaultdict(int)
        now = time.time()
        last_5_minutes = now - 300
        
        # Считаем ошибки за последние 5 минут
        for error in self.metrics['errors']:
            if error['timestamp'] > last_5_minutes:
                error_counts[error['error_type']] += 1
                
        # Алерт при более чем 5 ошибок одного типа за 5 минут
        if error_counts[error_type] > 5:
            self._add_alert('HIGH_ERROR_RATE', 
                          f'Высокая частота ошибок {error_type}: {error_counts[error_type]} за 5 мин')
            
    def _add_alert(self, alert_type, message):
        """Добавляет алерт"""
        alert = {
            'timestamp': time.time(),
            'type': alert_type,
            'message': message,
            'resolved': False
        }
        
        self.alerts.append(alert)
        logging.warning(f"ALERT [{alert_type}]: {message}")
        
        # Отправляем в систему мониторинга
        self._send_to_monitoring_system(alert)
        
    def get_active_alerts(self):
        """Возвращает активные алерты"""
        return [alert for alert in self.alerts if not alert['resolved']]
        
    def resolve_alert(self, alert_index):
        """Помечает алерт как решенный"""
        if 0 <= alert_index < len(self.alerts):
            self.alerts[alert_index]['resolved'] = True
