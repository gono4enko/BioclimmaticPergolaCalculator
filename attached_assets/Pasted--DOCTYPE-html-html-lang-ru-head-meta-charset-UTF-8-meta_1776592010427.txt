  const area=w*h;
  const glass=area*(gl==='transparent'?CFG.transparentGlass:CFG.tintedGlass)*(1+CFG.markup/100);
  const inst=area*CFG.install*(1+CFG.markup/100);
  const deliv=comp*(CFG.delivery/100)*(1+CFG.markup/100);
  return comp+glass+deliv+inst;
}

function drawPanoramaScheme(id, w, h, pc, dir, colorVal, glassType){
  var svg=document.getElementById('svg-'+id);
  if(!svg||!w||!h)return;
  var n=parseInt(pc)||3;
  var isCenter=(dir==='center'||n===6||n===8||n===10);
  var pC='#3a4048';
  if(colorVal==='ral9016')pC='#d0d0d0';
  else if(colorVal==='ral8028')pC='#5c3d1e';
  else if(colorVal==='ral7016')pC='#3a4048';
  else if(colorVal==='custom')pC='#777';
  var pD=(colorVal==='ral9016')?'#909090':(colorVal==='ral8028')?'#3a2410':'#111';
  var isTinted=(glassType==='tinted');
  var isBronze=(isTinted&&colorVal==='ral8028');
  var frameW=260;
  var aspect=h/w;
  var frameH=Math.round(frameW*aspect);
  if(frameH<60)frameH=60; if(frameH>230)frameH=230;
  var scale=frameW/(w*1000);
  var topPx   =Math.max(5, Math.min(14, Math.round(46*scale*frameW/100)));
  var botPx   =Math.max(3, Math.min(8,  Math.round(20*scale*frameW/100)));
  var sideOutPx=Math.max(4, Math.min(14, Math.round(45*scale*frameW/100)));
  var midPx   =Math.max(2, Math.min(5,  Math.round(30*scale*frameW/100)));
  var centerPx=Math.max(4, Math.min(12, Math.round(40*scale*frameW/100)));
  var padL=24,padT=14,railH=7;
  var VW=300,VH=padT+frameH+railH+22;
  svg.setAttribute('viewBox','0 0 '+VW+' '+VH);
  svg.setAttribute('height',VH);
  var fx=padL+4,fy=padT;
  var cGlass1,cGlass2;
  if(isBronze){cGlass1='#b8956a';cGlass2='#9a7548';}
  else if(isTinted){cGlass1='#8a9ea8';cGlass2='#6a8088';}
  else{cGlass1='#cce0f0';cGlass2='#a8c8e0';}
  var cRail='#b8c4cc',cDim='#8a9bbf',cArrow='#5a7a9a';
  var s='';
  s+='<rect x="'+fx+'" y="'+fy+'" width="'+frameW+'" height="'+topPx+'" fill="'+pC+'"/>';
  s+='<rect x="'+fx+'" y="'+(fy+topPx-1)+'" width="'+frameW+'" height="1" fill="'+pD+'"/>';
  s+='<rect x="'+fx+'" y="'+(fy+frameH-botPx)+'" width="'+frameW+'" height="'+botPx+'" fill="'+pC+'"/>';
  s+='<rect x="'+fx+'" y="'+(fy+frameH-botPx)+'" width="'+frameW+'" height="1" fill="'+pD+'"/>';
  s+='<rect x="'+fx+'" y="'+fy+'" width="'+sideOutPx+'" height="'+frameH+'" fill="'+pC+'"/>';
  s+='<rect x="'+(fx+sideOutPx-1)+'" y="'+fy+'" width="1" height="'+frameH+'" fill="'+pD+'"/>';
  s+='<rect x="'+(fx+frameW-sideOutPx)+'" y="'+fy+'" width="'+sideOutPx+'" height="'+frameH+'" fill="'+pC+'"/>';
  s+='<rect x="'+(fx+frameW-sideOutPx)+'" y="'+fy+'" width="1" height="'+frameH+'" fill="'+pD+'"/>';
  var innerX=fx+sideOutPx;
  var innerY=fy+topPx;
  var innerW=frameW-sideOutPx*2;
  var innerH=frameH-topPx-botPx;
  if(isCenter){
    var cx=innerX+innerW/2;
    s+='<rect x="'+(cx-centerPx/2)+'" y="'+innerY+'" width="'+centerPx+'" height="'+innerH+'" fill="'+pC+'"/>';
    s+='<rect x="'+(cx-centerPx/2)+'" y="'+innerY+'" width="2" height="'+innerH+'" fill="'+pD+'"/>';
    s+='<rect x="'+(cx+centerPx/2-2)+'" y="'+innerY+'" width="2" height="'+innerH+'" fill="'+pD+'"/>';
    s+='<rect x="'+(cx-0.5)+'" y="'+innerY+'" width="1" height="'+innerH+'" fill="'+pD+'" opacity="0.4"/>';
  }
  var totalPanels=n;
  var halfN=isCenter?totalPanels/2:totalPanels;
  var panelW, usablePerSection;
  if(isCenter){
    usablePerSection=(innerW-centerPx)/2;
    panelW=(usablePerSection-(halfN-1)*midPx)/halfN;
  } else {
    panelW=(innerW-(totalPanels-1)*midPx)/totalPanels;
  }
  var handleRatio=Math.min(0.85,Math.max(0.25,1-1000/(h*1000)));
  var handleAbsY=innerY+innerH*handleRatio;
  var handleLen, hVert, hT;
  if(isCenter){
    handleLen=Math.max(3, Math.min(10, Math.round(60*scale*frameW/100)));
    hVert=Math.max(2, Math.min(6, Math.round(40*scale*frameW/100)));
    hT=Math.max(0.8, Math.min(1.5, scale*frameW*0.006));
  } else {
    handleLen=10;hVert=4;hT=1.5;
  }
  for(var i=0;i<totalPanels;i++){
    var sectionIdx=isCenter?(i%(totalPanels/2)):i;
    var sectionOff=0;
    if(isCenter&&i>=totalPanels/2){sectionOff=(innerW-centerPx)/2+centerPx;}
    var px2=innerX+sectionOff+sectionIdx*(panelW+midPx);
    var gx=px2;
    var gw=panelW;
    var isFirstInSection=(sectionIdx===0);
    if(!isFirstInSection){
      s+='<rect x="'+(px2-midPx)+'" y="'+innerY+'" width="'+midPx+'" height="'+innerH+'" fill="'+pC+'"/>';
      s+='<rect x="'+(px2-midPx)+'" y="'+innerY+'" width="1" height="'+innerH+'" fill="'+pD+'"/>';
    }
    s+='<rect x="'+gx+'" y="'+innerY+'" width="'+gw+'" height="'+innerH+'" fill="'+cGlass1+'"/>';
    s+='<rect x="'+(gx+gw*0.5)+'" y="'+innerY+'" width="'+(gw*0.5)+'" height="'+innerH+'" fill="'+cGlass2+'" opacity="0.3"/>';
    if(isTinted||isBronze){
      var mc=isBronze?'rgba(255,220,150,0.2)':'rgba(220,235,240,0.22)';
      var mc2=isBronze?'rgba(255,220,150,0.12)':'rgba(220,235,240,0.15)';
      s+='<polygon points="'+(gx+gw*0.05)+','+innerY+' '+(gx+gw*0.22)+','+innerY+' '+(gx+gw*0.62)+','+(innerY+innerH)+' '+(gx+gw*0.45)+','+(innerY+innerH)+'" fill="'+mc+'"/>';
      s+='<polygon points="'+(gx+gw*0.55)+','+innerY+' '+(gx+gw*0.65)+','+innerY+' '+(gx+gw*0.98)+','+(innerY+innerH)+' '+(gx+gw*0.88)+','+(innerY+innerH)+'" fill="'+mc2+'"/>';
    } else {
      var bStep=Math.max(8,gw/Math.ceil(gw/12));
      var bOff=0;
      while(bOff<gw-4){
        var bx1=gx+bOff+3,by1=innerY+innerH*0.06;
        var bx2b=bx1-4,by2b=by1+innerH*0.14;
        if(bx1<gx+gw-2){
          s+='<line x1="'+bx1+'" y1="'+by1+'" x2="'+bx2b+'" y2="'+by2b+'" stroke="#7ab8d4" stroke-width="1" opacity="0.6" stroke-linecap="round"/>';
          if(by2b+innerH*0.07<innerY+innerH-4)
            s+='<line x1="'+(bx1-2)+'" y1="'+(by1+innerH*0.09)+'" x2="'+(bx2b-2)+'" y2="'+(by2b+innerH*0.09)+'" stroke="#7ab8d4" stroke-width="1" opacity="0.4" stroke-linecap="round"/>';
        }
        bOff+=bStep;
      }
    }
    var hy=handleAbsY;
    var isLeftmost=(i===0);
    var isRightmost=(i===totalPanels-1);
    var isLeftInner=isCenter&&(i===totalPanels/2-1);
    var isRightInner=isCenter&&(i===totalPanels/2);
    if(isLeftmost){
      s+='<line x1="'+gx+'" y1="'+(hy-hVert)+'" x2="'+gx+'" y2="'+(hy+hVert)+'" stroke="'+pD+'" stroke-width="'+(hT*1.5)+'" stroke-linecap="round"/>';
      s+='<line x1="'+gx+'" y1="'+hy+'" x2="'+(gx+handleLen)+'" y2="'+hy+'" stroke="'+pD+'" stroke-width="'+hT+'" stroke-linecap="round"/>';
    }
    if(isRightmost){
      var hxR=gx+gw;
      s+='<line x1="'+hxR+'" y1="'+(hy-hVert)+'" x2="'+hxR+'" y2="'+(hy+hVert)+'" stroke="'+pD+'" stroke-width="'+(hT*1.5)+'" stroke-linecap="round"/>';
      s+='<line x1="'+hxR+'" y1="'+hy+'" x2="'+(hxR-handleLen)+'" y2="'+hy+'" stroke="'+pD+'" stroke-width="'+hT+'" stroke-linecap="round"/>';
    }
    if(isLeftInner){
      var hxLI=gx+gw;
      s+='<line x1="'+hxLI+'" y1="'+(hy-hVert)+'" x2="'+hxLI+'" y2="'+(hy+hVert)+'" stroke="'+pD+'" stroke-width="'+(hT*1.5)+'" stroke-linecap="round"/>';
      s+='<line x1="'+hxLI+'" y1="'+hy+'" x2="'+(hxLI-handleLen)+'" y2="'+hy+'" stroke="'+pD+'" stroke-width="'+hT+'" stroke-linecap="round"/>';
    }
    if(isRightInner){
      s+='<line x1="'+gx+'" y1="'+(hy-hVert)+'" x2="'+gx+'" y2="'+(hy+hVert)+'" stroke="'+pD+'" stroke-width="'+(hT*1.5)+'" stroke-linecap="round"/>';
      s+='<line x1="'+gx+'" y1="'+hy+'" x2="'+(gx+handleLen)+'" y2="'+hy+'" stroke="'+pD+'" stroke-width="'+hT+'" stroke-linecap="round"/>';
    }
    var arY=innerY+innerH*0.72,arMid=gx+gw/2,arLen=Math.min(gw*0.38,16);
    var goRight=isCenter?(i>=totalPanels/2):(dir!=='left');
    var ax1=arMid-arLen/2,ax2=arMid+arLen/2;
    if(!goRight){var tmp=ax1;ax1=ax2;ax2=tmp;}
    s+='<line x1="'+ax1+'" y1="'+arY+'" x2="'+ax2+'" y2="'+arY+'" stroke="'+cArrow+'" stroke-width="0.8" opacity="0.6"/>';
    var tip=(goRight?-3:3);
    s+='<polyline points="'+(ax2+tip)+','+(arY-2.5)+' '+ax2+','+arY+' '+(ax2+tip)+','+(arY+2.5)+'" fill="none" stroke="'+cArrow+'" stroke-width="0.8" opacity="0.6"/>';
  }
  var railY=fy+frameH,rw=frameW+14,rx2=fx-7;
  s+='<rect x="'+rx2+'" y="'+railY+'" width="'+rw+'" height="'+railH+'" fill="'+cRail+'" rx="2"/>';
  s+='<rect x="'+(rx2+4)+'" y="'+(railY+1)+'" width="'+(rw-8)+'" height="2" fill="#fff" opacity="0.3" rx="1"/>';
  for(var j=0;j<totalPanels;j++){
    var slotX=rx2+8+j*(rw-16)/totalPanels;
    s+='<rect x="'+slotX+'" y="'+(railY+railH-3)+'" width="'+((rw-16)/totalPanels-2)+'" height="2" fill="#9aaabb" rx="1"/>';
  }
  var dimY=railY+railH+9;
  s+='<line x1="'+fx+'" y1="'+(dimY-3)+'" x2="'+fx+'" y2="'+(dimY+3)+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<line x1="'+(fx+frameW)+'" y1="'+(dimY-3)+'" x2="'+(fx+frameW)+'" y2="'+(dimY+3)+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<line x1="'+fx+'" y1="'+dimY+'" x2="'+(fx+frameW)+'" y2="'+dimY+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<text x="'+(fx+frameW/2)+'" y="'+(dimY+9)+'" text-anchor="middle" font-size="8" fill="'+cDim+'" font-family="Arial,sans-serif">'+(Math.round(w*1000))+' \u043c\u043c</text>';
  var hX=fx-6,hMid=fy+frameH/2;
  s+='<line x1="'+(hX-3)+'" y1="'+fy+'" x2="'+(hX+3)+'" y2="'+fy+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<line x1="'+(hX-3)+'" y1="'+(fy+frameH)+'" x2="'+(hX+3)+'" y2="'+(fy+frameH)+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<line x1="'+hX+'" y1="'+fy+'" x2="'+hX+'" y2="'+(fy+frameH)+'" stroke="'+cDim+'" stroke-width="0.8"/>';
  s+='<text x="'+(hX-4)+'" y="'+(hMid+3)+'" text-anchor="middle" font-size="8" fill="'+cDim+'" font-family="Arial,sans-serif" transform="rotate(-90,'+(hX-4)+','+hMid+')">'+(Math.round(h*1000))+' \u043c\u043c</text>';
  svg.innerHTML=s;
}

window.removeOp=function(id){
  const c=document.getElementById('openings-container');
  if(c.children.length<=1)return;
  c.removeChild(document.getElementById(id));
  renumber();
};

function renumber(){
  const cards=document.querySelectorAll('#openings-container .opening-card');
  cards.forEach(function(c,i){
    var b=c.querySelector('.op-num');
    var l=c.querySelector('.opening-label');
    if(b)b.textContent=i+1;
    if(l)l.textContent='\u041f\u0440\u043e\u0451\u043c '+(i+1);
  });
}

window.onFieldChange=function(id){
  const card=document.getElementById(id);
  if(!card)return;
  const w=parseFloat(card.querySelector('.w-inp').value)||0;
  const h=parseFloat(card.querySelector('.h-inp').value)||0;
  const pcSel=card.querySelector('.pc-sel');
  const dirSel=card.querySelector('.dir-sel');
  if(w>0&&h>0){
    const mp=minPanels(w,h);
    Array.from(pcSel.options).forEach(function(o){o.disabled=parseInt(o.value)<mp;});
    if(parseInt(pcSel.value)<mp){
      const first=Array.from(pcSel.options).find(function(o){return !o.disabled;});
      if(first)pcSel.value=first.value;
    }
    const pc=parseInt(pcSel.value);
    var centerOpt=Array.from(dirSel.options).find(function(o){return o.value==='center';});
    var isEven=(pc%2===0)&&pc>=4;
    if((w>=6&&pc>=6)||pc>=8){
      dirSel.value='center';dirSel.disabled=true;
    } else if(!isEven){
      if(centerOpt)centerOpt.disabled=true;
      if(dirSel.value==='center')dirSel.value='right';
      dirSel.disabled=false;
    } else {
      if(centerOpt)centerOpt.disabled=false;
      dirSel.disabled=false;
    }
    let warn=card.querySelector('.weight-warn');
    if(pc===2){
      const panelW2=w/2;
      const wt=panelW2*h*CFG.panelKg;
      if(wt>CFG.maxKg){
        pcSel.value='3';
        if(!warn){
          warn=document.createElement('div');
          warn.className='weight-warn';
          warn.style.cssText='font-size:12px;color:#e8622a;margin-top:8px;padding:7px 10px;background:#fff4f0;border-radius:6px;border:1px solid #ffd4c2';

(function(){
  const inp=document.getElementById('leadPhone');
  if(!inp)return;
  function getDigits(val){
    let d=val.replace(/\D/g,'');
    if(d.startsWith('8'))d='7'+d.slice(1);
    if(d.length>0&&!d.startsWith('7'))d='7'+d;
    return d.slice(0,11);
  }
  function format(digits){
    if(!digits)return '';
    let f='+7';
    if(digits.length>1)f+=' ('+digits.slice(1,Math.min(4,digits.length));
    if(digits.length>=4)f+=') '+digits.slice(4,Math.min(7,digits.length));
    if(digits.length>=7)f+='-'+digits.slice(7,Math.min(9,digits.length));
    if(digits.length>=9)f+='-'+digits.slice(9,11);
    return f;
  }
  inp.addEventListener('focus',function(){
    if(!this.value)this.value='+7 (';
    const len=this.value.length;
    setTimeout(()=>this.setSelectionRange(len,len),0);
  });
  inp.addEventListener('blur',function(){
    const d=getDigits(this.value);
    if(d.length<=1)this.value='';
  });
  inp.addEventListener('keydown',function(e){
    if(e.key==='Backspace'){
      e.preventDefault();
      const d=getDigits(this.value);
      if(d.length<=1){this.value='';return;}
      this.value=format(d.slice(0,-1));
      this.setSelectionRange(this.value.length,this.value.length);
      this.classList.remove('err');
    } else if(e.key==='Delete'){
      e.preventDefault();
    } else if(e.key==='Enter'){
      e.preventDefault();window.submitLead();
    }
  });
  inp.addEventListener('input',function(e){
    if(e.inputType==='deleteContentBackward'||e.inputType==='deleteContentForward')return;
    const d=getDigits(this.value);
    if(!d){this.value='';return;}
    this.value=format(d);
    this.setSelectionRange(this.value.length,this.value.length);
    this.classList.remove('err');
  });
})();
